def solve(n):
    if n % 2 == 0:
        n=n/2
    else:
        n= 3*n + 1
        # array
    return n

def cycle(num1):    
    
    if num1 == 1:
        return 1 # Base case testing
    
    # List.
    arr = [] 
    # Initialize.
    n=num1 
    
    #appending the start
    arr.append(n)
    # Update
    n = solve(n) 
    
    while(n>1):
        n= solve(n)
        arr.append(n)
    arr.append(1)
    return(len(arr))
def main():
    import numpy as np
    # Scanning the values
    num1 = int(input())
    num2 = int(input())
    
    # initializing the array
    arr = []
    
    # looping through the range of elements
    for i in range(num1,num2):
        # Appending
        arr.append(cycle(i))
    np.asarray(arr)
    print(num1,num2,np.max(arr),'\n')

main()
